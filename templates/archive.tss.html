{ LAYOUT = 'layout.tss.html' TITLE = 'Archive' posts = list_content('../content/published', {suffix: '.md'}) for post in posts post.year = post.published | date('%Y') post.month = post.published | date('%m') end for }

Posts by month

    {for group in posts | group_by(.year) | sort_by_desc(fn g -> g[0].year)}
  • {group[0].year} ({length(group)})
      {for subgroup in group | group_by(.month) | sort_by_desc(fn g -> g[0].month)}
    • {subgroup[0].published | date('%B')} ({length(subgroup)})
      • {end for}
    • {end for}

Posts by tag

    {for group in posts | flat_map(.tags) | group_by(fn x -> x) | sort_by(fn g -> g[0])}
  • {group[0]} ({length(group)})
    • {end for}
